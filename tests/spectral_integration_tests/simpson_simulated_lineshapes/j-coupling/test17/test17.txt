spinsys {
  channels 29Si
  nuclei 29Si 1H
  shift 1 0 6p 0.5 0 0 0
  shift 2 0 0 0 0 0 0
  jcoupling 1 2 1200 100 0.5 50 20 90
}

par {
  crystal_file     zcw28656
  variable dw      200

  sw               1e6/dw
  np               1024
  proton_frequency 400e6
  start_operator   I1x
  detect_operator  I1p
  method           direct
  gamma_angles     1
  spin_rate        0
  verbose          1101
}

proc pulseq {} {
  global par

  for {set i 0} {$i < $par(np)} {incr i} {
    acq
    delay $par(dw)
  }

}

proc main {} {
  global par

  set f [fsimpson]
  faddlb $f 10 0
  fsave $f $par(name).fid
}
